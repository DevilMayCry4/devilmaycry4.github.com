date: 2014-03-03 11:56:50
tags: IOS		
---

- (void)method:(NSString *)arg,...
{
    va_list list;
    va_start(list, arg);
    while (YES)
    {
        NSString *value = va_arg(list, NSString*);
        if (value)
        {
            NSLog(@"value %@",value);
        }
        else
        {
            break;
        }
    }
    va_end(list);
}
